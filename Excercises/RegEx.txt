# 1. Simplify these regular expressions
# /a**/ /a+*/ /a.*/ /(ab)?(ab)*/  /(a*|b*)+/  /ab?|ab/

 /a**/ => /.*/
 /a+*/ => /.*/
 /a.*/ => /a/
 /(ab)?(ab)*/ => /(ab)*/
 /(a*|b*)+/ => /(a|b)*/
 /ab?|ab/ => /ab?/
